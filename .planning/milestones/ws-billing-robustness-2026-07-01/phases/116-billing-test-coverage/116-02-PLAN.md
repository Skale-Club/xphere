```typescript
export type MeterReason = 'copilot_turn'

export async function meterDebit(
  orgId: string,
  reason: MeterReason,
  costUsd: number,
  refId?: string | null,
): Promise<{ allowed: boolean; balanceAfter: number }>
// Calls: supabase.rpc('debit_copilot_credits', { p_org_id, p_amount_usd, p_run_id: refId ?? null, p_reason: reason })
// If costUsd <= 0: short-circuits, returns { allowed: true, balanceAfter: 0 } WITHOUT calling rpc.
// On rpc error (thrown or `error` field set): catches, logs, returns { allowed: true, balanceAfter: 0 } (fails OPEN).
// On success: returns { allowed: res.allowed ?? true, balanceAfter: Number(res.balance_after ?? 0) }.

export async function grantCopilot(
  orgId: string,
  amountUsd: number,
  kind: 'topup' | 'grant',

// Calls: supabase.rpc('credit_copilot_credits', { p_org_id, p_amount_usd: amountUsd, p_kind: kind, p_ref: ref ?? null, p_note: note ?? null })
// On rpc error: THROWS `new Error('grantCopilot failed: ' + error.message)` (does NOT fail open — different contract from meterDebit).
// On success: returns Number(data ?? 0).

export async function resetCopilotForPeriod(
  orgId: string,
  includedUsd: number,
  periodEnd?: string | null,
): Promise<number>
// Calls: supabase.rpc('reset_copilot_credits', { p_org_id, p_included_usd: includedUsd, p_period_end: periodEnd ?? null })
// On rpc error: THROWS `new Error('resetCopilotForPeriod failed: ' + error.message)`.
// On success: returns Number(data ?? 0).
```

The actual SQL for `debit_copilot_credits` (migration 1225_metering_reason.sql) — this is the dual-bucket draw-down math the mocked `.rpc()` return value must SIMULATE (not execute):
```sql
v_take_inc := greatest(0, least(v_inc, p_amount_usd));  -- take from included first, capped at what's available
v_inc := v_inc - v_take_inc;
v_top := v_top - (p_amount_usd - v_take_inc);            -- remainder (possibly all of it) comes from topup, can go negative
v_total := v_inc + v_top;
-- ledger insert: (org_id, 'debit', -p_amount_usd, v_total, p_run_id, p_reason)
RETURN jsonb_build_object('allowed', v_total > 0, 'balance_after', v_total);
```
Worked example used in RESEARCH.md: included=3, topup=10, debit amount=5 → take_inc=min(3,5)=3, inc becomes 0, top becomes 10-(5-3)=8, total=8, allowed=true.
Insufficient-balance example: included=2, topup=1, debit amount=5 → take_inc=min(2,5)=2, inc becomes 0, top becomes 1-(5-2)=-2, total=-2, allowed=false (v_total > 0 is false). The debit still fully applied (ledger amount_usd is still -5, balance_after is -2) — it is NOT clamped or refused.
