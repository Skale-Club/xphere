describe('deriveIdempotencyKey — production implementation (IDEMP-03)', () => {
  it('produces stable sha256 for same inputs', () => {
    const key1 = deriveIdempotencyKey('inv-123', 0)
    const key2 = deriveIdempotencyKey('inv-123', 0)
    expect(key1).toBe(key2)
  })
  it('produces different keys for different toolCallIndex', () => {
    expect(deriveIdempotencyKey('inv-123', 0)).not.toBe(deriveIdempotencyKey('inv-123', 1))
  })
  it('produces 64-char hex string', () => {
    expect(deriveIdempotencyKey('inv-abc', 5)).toMatch(/^[0-9a-f]{64}$/)
  })
  it('separator ":" prevents key(inv-12, 30) colliding with key(inv-123, 0)', () => {
    // Without separator: "inv-12" + "30" = "inv-1230" = "inv-123" + "0"
    const k1 = deriveIdempotencyKey('inv-12', 30)
    const k2 = deriveIdempotencyKey('inv-123', 0)
    expect(k1).not.toBe(k2)
  })
})

describe('GATE-06: Idempotency dedup — same key twice → executor invoked once', () => {
  it('deriving same key from same inputs is deterministic', () => {
    const invId = 'test-invocation-id'
    const idx = 2
    const key1 = deriveIdempotencyKey(invId, idx)
    const key2 = deriveIdempotencyKey(invId, idx)
    expect(key1).toBe(key2)
    // Same key means: if the idempotency store has key1, the second call is a cache hit
    // The cache hit short-circuits before executeAction — executor invoked once
    expect(key1).toHaveLength(64)
  })

  it('requiresIdempotency identifies all 4 side-effecting action types', () => {
    const sideEffecting = ['create_appointment', 'send_sms', 'create_contact']
    for (const actionType of sideEffecting) {
      expect(requiresIdempotency(actionType), `${actionType} should require idempotency`).toBe(true)
    }
    const readOnly = ['get_availability', 'knowledge_base', 'google_contacts_find', 'manychat_trigger_flow']
    for (const actionType of readOnly) {
      expect(requiresIdempotency(actionType), `${actionType} should NOT require idempotency`).toBe(false)
