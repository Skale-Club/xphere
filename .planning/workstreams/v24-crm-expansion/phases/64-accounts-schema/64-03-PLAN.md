```ts
import { describe, it, expect, beforeAll, afterAll } from 'vitest'
import { Client } from 'pg'

const DB_URL = process.env.SUPABASE_DB_URL ?? process.env.DATABASE_URL

let pg: Client | null = null

beforeAll(async () => {
  if (!DB_URL) {
    console.warn('[suite] SUPABASE_DB_URL/DATABASE_URL not set — will skip.')
    return
  }
  pg = new Client({ connectionString: DB_URL })
  await pg.connect()
}, 30000)

afterAll(async () => {
  if (pg) { await pg.end(); pg = null }
})

const dbSuite = DB_URL ? describe : describe.skip

dbSuite('TABLE has RLS enabled + canonical policy', () => {
  // 1. relrowsecurity check
  const relRes = await pg!.query<{ relrowsecurity: boolean }>(
    `SELECT c.relrowsecurity FROM pg_class c
       JOIN pg_namespace n ON n.oid = c.relnamespace
      WHERE n.nspname = 'public' AND c.relname = $1`,
    [table],
  )
  expect(relRes.rows[0].relrowsecurity).toBe(true)

  // 2. policy with get_current_org_id() in USING
  const polRes = await pg!.query(
    `SELECT polname, pg_get_expr(polqual, polrelid) AS using_expr
       FROM pg_policy
      WHERE polrelid = (SELECT c.oid FROM pg_class c
        JOIN pg_namespace n ON n.oid = c.relnamespace
        WHERE n.nspname='public' AND c.relname=$1)`,
    [table],
  )
  expect(polRes.rows.some(r => r.using_expr?.includes('get_current_org_id'))).toBe(true)
})
```

**Test 1 — ACC-19 schema-layer: accounts has RLS enabled + canonical policy.**

**Test 2 — ACC-15: CHECK constraint rejects orphan opportunities.**

Use the `pgSuite`. Connect as the postgres superuser (the connection in `DB_URL` is the service-role / direct DB URL, which bypasses RLS but DOES NOT bypass CHECK constraints — see Postgres docs: CHECK is row-level integrity, not access control).

Inside `pgSuite('ACC-15: opportunities CHECK constraint', () => { ... })`:

- `it('opp_has_contact_or_account constraint exists on public.opportunities', ...)` — query `pg_constraint` where `conname='opp_has_contact_or_account' AND conrelid='public.opportunities'::regclass`. Expect `rowCount === 1`. Also assert `pg_get_constraintdef(oid)` contains the substring `contact_id IS NOT NULL OR account_id IS NOT NULL`.
- `it('inserting an opportunity with both contact_id NULL and account_id NULL is rejected', ...)` — seed prerequisites: create a throwaway org via service-role pg client, create a pipeline + stage under that org (use raw SQL since the test connection bypasses RLS), then attempt:
  ```sql
  INSERT INTO public.opportunities (org_id, pipeline_id, stage_id, title)
  VALUES ($1, $2, $3, 'test-orphan-' || $4);
  ```
  Wrap in try/catch and expect the error message to contain either `opp_has_contact_or_account` or `violates check constraint`. Then clean up the seeded org (cascades will remove the pipeline/stage).

  **Important — sentinel cleanup:** Use a deterministic unique suffix for any seeded rows (e.g. `'acc-schema-' + crypto.randomUUID()`) so re-running the test never piles up orphan org rows. Wrap in a `try { ... } finally { await cleanup() }` block.

---

**Test 3 — ACC-14: data migration block is idempotent (zero new rows on second run).**

Use the `pgSuite`. Inside `pgSuite('ACC-14: contacts.company → accounts data migration is idempotent', () => { ... })`:

- `it('re-running the data-migration block produces zero new accounts rows', async () => { ... })`:
  1. Seed a throwaway org via direct pg query.
  2. Insert 5 contacts under that org with `company` values: `'Acme Corp'`, `'  Acme Corp  '` (trim/dup test), `'Beta LLC'`, `''` (empty — must be ignored), `NULL` (must be ignored). Expected distinct accounts after first run = 2 (Acme Corp, Beta LLC).
  3. **Run the migration block** verbatim from `supabase/migrations/064_accounts.sql` section 9 (the CTE + INSERT + UPDATE), scoped to this org by adding `AND org_id = $orgId` to the CTE's WHERE clause. Capture `accounts` count for this org → expect 2.
  4. **Run the same block a second time** with the same org scope. Capture `accounts` count for this org again → expect 2 (no new rows).
  5. Verify all 3 non-empty contacts (the two "Acme" + the one "Beta") have `account_id` populated and pointing to the right account by `lower(name)`.
  6. Cleanup the throwaway org.

  Note: scoping the CTE to one org keeps the test isolated and re-runnable without bleeding across real data.

**Test 4 — ACC-19 cross-org: account inserted in org A is invisible from org B's authenticated client.**

Use the `fullSuite` (requires both DB_URL and Supabase env vars). Follow the seed pattern from `tests/rls-isolation.test.ts` lines 34–80: create two orgs, two users, give each user `admin` membership in their org via `org_members` table, sign in each user with the anon client.

Inside `fullSuite('ACC-19: accounts cross-org isolation', () => { ... })`:

- `it('account inserted under org A is not visible to a user signed into org B', async () => { ... })`:
  1. With the service-role admin client, insert an account into org A with `name = 'CrossOrg Test ' + suffix`. Capture its id.
  2. With the anon client signed in as user B (org B member), query `from('accounts').select('id, name').eq('id', accountAId).maybeSingle()`. Assert `data === null` (RLS hides it).
  3. With the anon client signed in as user A, query the same row. Assert `data?.id === accountAId`.
  4. Cleanup: service-role deletes both orgs (cascades).

Note: the active-org cookie pattern (`vo_active_org`) used in production server actions is set by `get_current_org_id()` which falls back to "first membership" when no explicit selection is stored — that's what makes this test work without manually setting the active org.
