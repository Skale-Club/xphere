  ```typescript
  let cachedToken: string | null = null
  let cachedExp = 0   // unix seconds
  function b64urlToJson(b64: string): { exp?: number } | null {
    try { const pad = b64.length % 4 === 0 ? '' : '='.repeat(4 - (b64.length % 4))
      return JSON.parse(atob(b64.replace(/-/g, '+').replace(/_/g, '/') + pad)) } catch { return null }
  }
  function setContext(token: string): void {
    const p = b64urlToJson(token.split('.')[0] ?? '')
    cachedToken = token
    cachedExp = typeof p?.exp === 'number' ? p.exp : Math.floor(Date.now() / 1000) + 60
  }
  async function ensureContext(): Promise<string | null> {
    if (!contextEndpoint) return null
    const now = Math.floor(Date.now() / 1000)
    if (cachedToken && cachedExp > now + 5) return cachedToken
    try {
      const res = await fetch(contextEndpoint, { credentials: 'same-origin' })   // SAME-ORIGIN — never apiBase
      if (!res.ok) return null
      const { token } = await res.json() as { token?: string }
      if (!token) return null
      setContext(token)
      return cachedToken
    } catch { return null }
  }
  ```
