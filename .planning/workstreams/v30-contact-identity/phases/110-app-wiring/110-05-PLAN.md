```ts
export async function getConflictCount(
  supabase: SupabaseClient<Database>,
): Promise<number> {
  const { count } = await supabase
    .from('contacts')
    .select('id', { count: 'exact', head: true })
    .eq('identity_status', 'merge_conflict')
  return count ?? 0
}
```
