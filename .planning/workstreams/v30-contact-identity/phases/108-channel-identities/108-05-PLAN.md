       ```ts
       import { describe, it, expect, beforeAll, afterAll, vi } from 'vitest'
       import { Client } from 'pg'
       import { readFileSync } from 'node:fs'
       import { resolve } from 'node:path'
       import { createClient } from '@supabase/supabase-js'

       vi.mock('server-only', () => ({}))

       // Import AFTER server-only mock
       const { findByChannelIdentity, attachChannelIdentity } = await import('@/lib/contacts/server')

       function readDbUrl(): string | null {
         try {
           const env = readFileSync(resolve(process.cwd(), '.env.local'), 'utf8')
           const m = env.match(/^DATABASE_URL=(.+)$/m)
           return m ? m[1].trim().replace(/^["']|["']$/g, '') : null
         } catch { return null }
       }

       function readSupabaseConfig(): { url: string; serviceKey: string } | null {
         try {
           const env = readFileSync(resolve(process.cwd(), '.env.local'), 'utf8')
           const url = env.match(/^NEXT_PUBLIC_SUPABASE_URL=(.+)$/m)?.[1].trim().replace(/^["']|["']$/g, '')
           const key = env.match(/^SUPABASE_SERVICE_ROLE_KEY=(.+)$/m)?.[1].trim().replace(/^["']|["']$/g, '')
           return url && key ? { url, serviceKey: key } : null
         } catch { return null }
       }

       const dbUrl = readDbUrl()
       const sbCfg = readSupabaseConfig()
       const skipReason = !dbUrl || !sbCfg ? 'DATABASE_URL or Supabase config missing — skip integration tests' : null

       describe.skipIf(skipReason)('contact_channel_identities (Phase 108)', () => {
         // Track ids for cleanup
         const createdContactIds: string[] = []
         const createdIdentityIds: string[] = []
         let orgId: string
         let admin: Client

         beforeAll(async () => {
           admin = new Client({ connectionString: dbUrl! })
           await admin.connect()
           const { rows } = await admin.query('SELECT id FROM public.organizations LIMIT 1')
           orgId = rows[0].id
         })

         afterAll(async () => {
           if (createdIdentityIds.length) {
             await admin.query('DELETE FROM public.contact_channel_identities WHERE id = ANY($1::uuid[])', [createdIdentityIds])
           }
           if (createdContactIds.length) {
             // Cascade removes any remaining identity rows automatically
             await admin.query('DELETE FROM public.contacts WHERE id = ANY($1::uuid[])', [createdContactIds])
           }
           await admin.end()
         })

         async function insertContact(extra: Record<string, unknown> = {}): Promise<string> {
           const { rows } = await admin.query(
             `INSERT INTO public.contacts (org_id, name, source)
              VALUES ($1, $2, 'manual') RETURNING id`,
             [orgId, 'phase108-test-' + Math.random().toString(36).slice(2, 8)],
           )
           const id = rows[0].id
           createdContactIds.push(id)
           if (Object.keys(extra).length) {
             const setClauses = Object.keys(extra).map((k, i) => `${k} = $${i + 2}`).join(', ')
             await admin.query(
               `UPDATE public.contacts SET ${setClauses} WHERE id = $1`,
               [id, ...Object.values(extra)],
             )
           }
           return id
         }

         it('UNIQUE (org_id, provider, external_id) — second insert raises 23505', async () => {
           const c1 = await insertContact()
           const c2 = await insertContact()
           const externalId = 'race-' + Math.random().toString(36).slice(2, 10)

           // First insert succeeds
           const r1 = await admin.query(
             `INSERT INTO public.contact_channel_identities (org_id, contact_id, provider, external_id)
              VALUES ($1, $2, 'webchat', $3) RETURNING id`,
             [orgId, c1, externalId],
           )
           createdIdentityIds.push(r1.rows[0].id)

           // Second insert (different contact, same identity tuple) must fail with 23505
           let err: unknown
           try {
             await admin.query(
               `INSERT INTO public.contact_channel_identities (org_id, contact_id, provider, external_id)
                VALUES ($1, $2, 'webchat', $3)`,
               [orgId, c2, externalId],
             )
           } catch (e) { err = e }
           expect(err).toBeDefined()
           expect((err as { code?: string }).code).toBe('23505')
         })

         it('ON DELETE CASCADE — deleting contact removes its identities', async () => {
           const c = await insertContact()
           const externalId = 'cascade-' + Math.random().toString(36).slice(2, 10)
           await admin.query(
             `INSERT INTO public.contact_channel_identities (org_id, contact_id, provider, external_id)
              VALUES ($1, $2, 'telegram', $3)`,
             [orgId, c, externalId],
           )
           // Remove c from cleanup list — we delete it here intentionally
           const idx = createdContactIds.indexOf(c)
           if (idx >= 0) createdContactIds.splice(idx, 1)

           await admin.query('DELETE FROM public.contacts WHERE id = $1', [c])

           const { rows } = await admin.query(
             `SELECT count(*)::int AS n FROM public.contact_channel_identities
              WHERE org_id = $1 AND provider = 'telegram' AND external_id = $2`,
             [orgId, externalId],
           )
           expect(rows[0].n).toBe(0)
         })

         it('findByChannelIdentity resolves merged_into chain to live survivor', async () => {
           const survivor = await insertContact()
           const archived = await insertContact({
             identity_status: 'archived_duplicate',
             merged_into_contact_id: survivor,
           })
           const externalId = 'chain-' + Math.random().toString(36).slice(2, 10)
           const r = await admin.query(
             `INSERT INTO public.contact_channel_identities (org_id, contact_id, provider, external_id)
              VALUES ($1, $2, 'instagram', $3) RETURNING id`,
             [orgId, archived, externalId],
           )
           createdIdentityIds.push(r.rows[0].id)

           const sb = createClient(sbCfg!.url, sbCfg!.serviceKey)
           const hit = await findByChannelIdentity(sb, orgId, 'instagram', externalId)
           expect(hit).not.toBeNull()
           expect(hit!.contact_id).toBe(survivor)
         })

         it('attachChannelIdentity is idempotent — second call returns same contact_id, no duplicate row', async () => {
           const c = await insertContact()
           const externalId = 'idem-' + Math.random().toString(36).slice(2, 10)
           const sb = createClient(sbCfg!.url, sbCfg!.serviceKey)

           const r1 = await attachChannelIdentity(sb, orgId, c, 'whatsapp', externalId)
           const r2 = await attachChannelIdentity(sb, orgId, c, 'whatsapp', externalId)
           expect(r1?.contact_id).toBe(c)
           expect(r2?.contact_id).toBe(c)

           const { rows } = await admin.query(
             `SELECT id FROM public.contact_channel_identities
              WHERE org_id = $1 AND provider = 'whatsapp' AND external_id = $2`,
             [orgId, externalId],
           )
           expect(rows.length).toBe(1)
           createdIdentityIds.push(rows[0].id)
         })
       })
       ```
