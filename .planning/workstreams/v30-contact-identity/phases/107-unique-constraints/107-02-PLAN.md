    ```ts
    import type { SupabaseClient } from '@supabase/supabase-js'
    import type { Database } from '@/types/database'
    import { normalisePhone, normaliseEmail } from '@/lib/contacts/zod-schemas'

    /**
     * Look up a live contact by phone within an org.
     *
     * Filters out archived_duplicate rows (matches partial UNIQUE index predicate).
     * Pass an authenticated Supabase client; RLS still applies. The orgId argument
     * is required because some call sites (server actions) already hold it and
     * we avoid a round-trip; webhook handlers usually look it up first.
     *
     * Returns `null` if no live contact matches or phone normalises to null.
     */
    export async function findByPhone(
      supabase: SupabaseClient<Database>,
      orgId: string,
      phone: string | null | undefined,
    ): Promise<{ id: string } | null> {
      const normalized = normalisePhone(phone ?? null)
      if (!normalized) return null
      const { data } = await supabase
        .from('contacts')
        .select('id')
        .eq('org_id', orgId)
        .eq('phone_e164', normalized)
        .neq('identity_status', 'archived_duplicate')
        .maybeSingle()
      return data ?? null
    }

    /**
     * Look up a live contact by email within an org.
     *
     * Mirror of findByPhone for the email_normalized partial index.
     */
    export async function findByEmail(
      supabase: SupabaseClient<Database>,
      orgId: string,
      email: string | null | undefined,
    ): Promise<{ id: string } | null> {
      const normalized = normaliseEmail(email ?? null)
      if (!normalized) return null
      const { data } = await supabase
        .from('contacts')
        .select('id')
        .eq('org_id', orgId)
        .eq('email_normalized', normalized)
        .neq('identity_status', 'archived_duplicate')
        .maybeSingle()
      return data ?? null
    }
    ```

    Notes:
    - If `src/lib/contacts/server.ts` does NOT already import the `SupabaseClient` and `Database` types, add the imports at the top.
    - Keep the `'server-only'` import.
    - Both helpers take a passed-in client so webhook handlers (plan 03) can pass their service-role client without re-instantiating.
