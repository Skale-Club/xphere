<planning_decisions_locked>
- **Anti-loop pattern: claim-first.** Insert the `ghl_reengagement_sent` row BEFORE the Twilio call using the implicit `ON CONFLICT DO NOTHING` semantics of `.insert(...).select('id').single()` (when `.single()` returns `null` data due to no insert, treat as conflict → skip). On Twilio failure, DELETE the just-inserted row. This is more robust than send-then-record against T-32-04 (workflow_dispatch racing the cron).
- **BATCH_LIMIT enforcement.** The runner stops dispatching after `cfg.batchLimit` successful sends. Anything in the GHL list past that count is silently dropped (not counted in `skipped` — they'll be picked up next run). The locked product default in Plan 04 is 20 (source of truth: STATE.md `### v1.9 Scope` + REQUIREMENTS.md REENG-16, revised 2026-05-15).
- **Twilio integration pre-flight.** Per Assumption A4, before any dispatch the runner SELECTs `integrations` for `(organization_id=cfg.orgId, provider='twilio', is_active=true)` and asserts `row.id === cfg.twilioIntegrationId`. Mismatch → throw a clear error so the route returns 500.
- **PII redaction in logs.** `request_payload.phone_masked = '***' + last4`. `request_payload.message_rendered_first40 = rendered.substring(0, 40)`. Full phone never enters `action_logs`.
- **Promise.allSettled for parallel dispatch.** Process the batch in parallel (not serial); one Twilio failure does not block others. Tradeoff: rate-limit risk from Twilio. Acceptable at batchLimit=20 (well under Twilio's default 1msg/sec from one number; if rate-limited, individual sends fail and are counted in `failed`).
- **`vapi_call_id` synthetic value.** `'cron:ghl-reengagement:' + cfg.runStartedAtIso`. Free-text TEXT NOT NULL column (`002_action_engine.sql:101`) accepts this.
- **`tool_config_id: null`.** This dispatch is not bound to a `tool_configs` row. The column is nullable via `ON DELETE SET NULL`.
- **Phone validation.** Strict: `phone.startsWith('+')` AND `/^\+\d{8,15}$/.test(phone)`. Anything else → counted as `skipped` with `reason: 'invalid_phone_format'` logged.
- **JS-side date guard.** Even if GHL ignores `updatedBefore`, we filter `updatedAt > thresholdCutoff` in JS before the dispatch loop. Pitfall 1 defense-in-depth.

```typescript
// src/lib/automations/ghl-reengagement/runner.ts
// Phase 32 (v1.9): GHL Lost-Lead Reengagement SMS Runner
// Pure orchestration: list → filter → claim → dispatch → log.
// Pattern source: 32-RESEARCH.md Patterns 1-5, 7; Security Domain "claim-first".
//
// REUSE — do NOT reimplement:
//   - listOpportunities (Plan 02)
//   - renderMessage (Plan 02)
//   - sendSms (v1.8)
//   - logAction (v1.0)
//   - service-role supabase client (caller's responsibility — runner accepts it)

import type { SupabaseClient } from '@supabase/supabase-js'
import type { Database, Json } from '@/types/database'
import type { GhlCredentials } from '@/lib/ghl/client'
import { listOpportunities, type GhlOpportunity } from '@/lib/ghl/list-opportunities'
import { renderMessage } from './render-template'
import { sendSms } from '@/lib/twilio/send-sms'
import { logAction } from '@/lib/action-engine/log-action'

export interface RunnerConfig {
  orgId: string
  locationId: string
  ghlCredentials: GhlCredentials
  twilioIntegrationId: string
  messageTemplate: string
  thresholdDays: number
  batchLimit: number
  runStartedAtIso: string
}

export interface RunnerError {
  ghl_contact_id: string
  error_message: string
}

export interface RunnerResult {
  processed: number
  sent: number
  skipped: number
  failed: number
  errors: RunnerError[]
}

const TOOL_NAME = 'ghl_reengagement_sms'
const E164_REGEX = /^\+\d{8,15}$/
const PHONE_MASK_PREFIX = '***'
const BODY_LOG_TRUNCATE = 40

function maskPhone(phone: string): string {
  const last4 = phone.slice(-4)
  return `${PHONE_MASK_PREFIX}${last4}`
}

function isE164(phone: string | null | undefined): boolean {
  return typeof phone === 'string' && E164_REGEX.test(phone)
}

function opportunityTimestamp(opp: GhlOpportunity): number | null {
  const raw = opp.updatedAt ?? opp.statusChangeDate
  if (!raw) return null
  const t = Date.parse(raw)
  return Number.isFinite(t) ? t : null
}

interface DispatchOutcome {
  ghl_contact_id: string
  result: 'sent' | 'skipped' | 'failed'
  skipReason?: 'in_anti_loop' | 'invalid_phone_format' | 'within_threshold' | 'conflict_on_claim'
  errorMessage?: string
}

export async function runReengagement(
  cfg: RunnerConfig,
  supabase: SupabaseClient<Database>,
): Promise<RunnerResult> {
  // ---- Pre-flight: Twilio integration assertion (Assumption A4) ----
  const { data: twilioRow, error: twilioErr } = await supabase
    .from('integrations')
    .select('id')
    .eq('organization_id', cfg.orgId)
    .eq('provider', 'twilio')
    .eq('is_active', true)
    .single()
  if (twilioErr || !twilioRow) {
    throw new Error('Twilio integration not found or inactive for this org')
  }
  if (twilioRow.id !== cfg.twilioIntegrationId) {
    throw new Error(
      `Twilio integration mismatch: env GHL_REENGAGEMENT_TWILIO_INTEGRATION_ID=${cfg.twilioIntegrationId} ` +
      `but org's active Twilio integration id=${twilioRow.id}`,
    )
  }

  // ---- 1. List Lost opportunities older than threshold ----
  const thresholdCutoff = new Date(Date.now() - cfg.thresholdDays * 24 * 60 * 60 * 1000)
  const opportunities = await listOpportunities(cfg.ghlCredentials, {
    status: 'lost',
    updatedBefore: thresholdCutoff,
    // batchLimit translates to GHL page limit (capped at 100 per GHL convention)
    limit: Math.min(cfg.batchLimit, 100),
  })

  // ---- 2. JS-side date guard (defense in depth — Pitfall 1, Assumption A2) ----
  const cutoffMs = thresholdCutoff.getTime()
  const dateFiltered = opportunities.filter(opp => {
    const t = opportunityTimestamp(opp)
    return t !== null && t < cutoffMs
  })

  const processed = dateFiltered.length

  if (processed === 0) {
    return { processed: 0, sent: 0, skipped: 0, failed: 0, errors: [] }
  }

  // ---- 3. Bulk anti-loop pre-filter (one SELECT, not N) ----
  const ghlContactIds = dateFiltered.map(o => o.contact.id)
  const { data: already } = await supabase
    .from('ghl_reengagement_sent')
    .select('ghl_contact_id')
    .eq('org_id', cfg.orgId)
    .in('ghl_contact_id', ghlContactIds)
  const alreadySet = new Set((already ?? []).map(r => r.ghl_contact_id))

  // ---- 4. Build per-contact outcomes for non-dispatch reasons ----
  const outcomes: DispatchOutcome[] = []
  const toDispatch: GhlOpportunity[] = []

  for (const opp of dateFiltered) {
    if (alreadySet.has(opp.contact.id)) {
      outcomes.push({ ghl_contact_id: opp.contact.id, result: 'skipped', skipReason: 'in_anti_loop' })
      continue
    }
    if (!isE164(opp.contact.phone)) {
      outcomes.push({ ghl_contact_id: opp.contact.id, result: 'skipped', skipReason: 'invalid_phone_format' })
      continue
    }
    toDispatch.push(opp)
  }

  // ---- 5. Enforce batchLimit on actual dispatches ----
  const batch = toDispatch.slice(0, cfg.batchLimit)

  // ---- 6. Claim-first + dispatch in parallel via Promise.allSettled ----
  await Promise.allSettled(
    batch.map(async opp => {
      const phone = opp.contact.phone as string  // guarded by isE164 above
      const startMs = Date.now()

      // Claim BEFORE send (ON CONFLICT DO NOTHING semantics)
      let claimedId: string | null = null
      try {
        const { data: claim } = await supabase
          .from('ghl_reengagement_sent')
          .insert({
            org_id: cfg.orgId,
            location_id: cfg.locationId,
            ghl_contact_id: opp.contact.id,
          })
          .select('id')
          .single()
        claimedId = claim?.id ?? null
      } catch {
        // PostgREST throws on UNIQUE conflict — treat as conflict
        claimedId = null
      }

      if (!claimedId) {
        outcomes.push({
          ghl_contact_id: opp.contact.id,
          result: 'skipped',
          skipReason: 'conflict_on_claim',
        })
        return
      }

      const rendered = renderMessage(cfg.messageTemplate, opp.contact.firstName)
      const logBase = {
        organization_id: cfg.orgId,
        tool_config_id: null,
        vapi_call_id: `cron:ghl-reengagement:${cfg.runStartedAtIso}`,
        tool_name: TOOL_NAME,
        request_payload: {
          ghl_contact_id: opp.contact.id,
          phone_masked: maskPhone(phone),
          message_rendered_first40: rendered.substring(0, BODY_LOG_TRUNCATE),
        } as Json,
      }

      try {
        const result = await sendSms(
          { to: phone, body: rendered },
          { organizationId: cfg.orgId, supabase },
        )
        const elapsed = Date.now() - startMs
        await logAction({
          ...logBase,
          status: 'success',
          execution_ms: elapsed,
          response_payload: { result } as Json,
          error_detail: null,
        }, supabase)
        outcomes.push({ ghl_contact_id: opp.contact.id, result: 'sent' })
      } catch (err) {
        // Twilio failure — roll back the claim
        const elapsed = Date.now() - startMs
        const message = err instanceof Error ? err.message : String(err)
        await supabase.from('ghl_reengagement_sent').delete().eq('id', claimedId)
        await logAction({
          ...logBase,
          status: 'error',
          execution_ms: elapsed,
          response_payload: {} as Json,
          error_detail: message,
        }, supabase)
        outcomes.push({ ghl_contact_id: opp.contact.id, result: 'failed', errorMessage: message })
      }
    }),
  )

  // ---- 7. Aggregate outcomes into RunnerResult ----
  let sent = 0, skipped = 0, failed = 0
  const errors: RunnerError[] = []
  for (const o of outcomes) {
    if (o.result === 'sent') sent++
    else if (o.result === 'skipped') skipped++
    else if (o.result === 'failed') {
      failed++
      errors.push({ ghl_contact_id: o.ghl_contact_id, error_message: o.errorMessage ?? 'unknown' })
    }
  }

  return { processed, sent, skipped, failed, errors }
}
```
