    ```typescript
    import { createServiceRoleClient } from '@/lib/supabase/admin'
    import { normalizeEmail } from '@/lib/auth/zod-schemas'

    export interface ReconcileResult {
      /** Number of org_members rows upserted in this call. */
      membershipsCreated: number
      /** Number of org_invites rows marked accepted_at in this call. */
      invitesAccepted: number
      /** Org to seed user_active_org / vo_active_org cookie with. Null if no invites matched. */
      firstOrgId: string | null
      /** Friendly name of firstOrg for the cookie. Null if no invites matched. */
      firstOrgName: string | null
    }

    /**
     * Reconciles a newly-authenticated user against pending org_invites.
     *
     * Contract:
     * - Uses service-role client (bypasses RLS — required because new user has no
     *   org_members row yet, so get_current_org_id() returns NULL → RLS would block
     *   every query). See RESEARCH Pitfall 4.
     * - Email is normalized internally (lowercase + trim) per D-42-10.
     * - For every pending, non-expired invite matching the email, UPSERTs an
     *   org_members row (onConflict: user_id,organization_id, ignoreDuplicates) and
     *   UPDATEs the invite's accepted_at. UPSERT makes this safe for:
     *     - D-42-11 (subsequent logins — no double-INSERT)
     *     - RESEARCH Pitfall 7 (re-invite after removal — no unique-constraint crash)
     * - Returns membershipsCreated/invitesAccepted counts + firstOrgId/Name for cookie seeding.
     * - If no invites match: returns counts=0, firstOrgId=null. Caller MUST sign user out
     *   and redirect to /login?error=not_invited (D-42-12).
     *
     * Non-atomicity acknowledgement: the loop is not transactional. If the second
     * invite's UPDATE fails after its INSERT succeeded, re-login will re-process
     * the still-pending invite (membership upsert is idempotent). See RESEARCH Open Q1.
     */
    export async function reconcileInvitesForUser(args: {
      userId: string
      email: string
    }): Promise<ReconcileResult> {
      const email = normalizeEmail(args.email)
      const admin = createServiceRoleClient()

      // 1. Find pending, non-expired invites for this email (RLS bypassed).
      const { data: invites, error: invitesError } = await admin
        .from('org_invites')
        .select('id, org_id, role')
        .eq('email', email)
        .is('accepted_at', null)
        .gt('expires_at', new Date().toISOString())

      if (invitesError) {
        // Surface error to caller (callback will redirect to /login?error=auth_callback_failed).
        throw new Error(`reconcileInvitesForUser: invite lookup failed: ${invitesError.message}`)
      }

      if (!invites || invites.length === 0) {
        return { membershipsCreated: 0, invitesAccepted: 0, firstOrgId: null, firstOrgName: null }
      }

      const now = new Date().toISOString()
      let membershipsCreated = 0
      let invitesAccepted = 0

      // 2. Loop. Not atomic but idempotent on re-run.
      for (const inv of invites) {
        const { error: memberError } = await admin
          .from('org_members')
          .upsert(
            { user_id: args.userId, organization_id: inv.org_id, role: inv.role },
            { onConflict: 'user_id,organization_id', ignoreDuplicates: true }
          )
        if (memberError) {
          throw new Error(`reconcileInvitesForUser: membership upsert failed for org ${inv.org_id}: ${memberError.message}`)
        }
        membershipsCreated += 1

        const { error: inviteError } = await admin
          .from('org_invites')
          .update({ accepted_at: now })
          .eq('id', inv.id)
        if (inviteError) {
          // Membership succeeded; invite update failed. Re-login will re-process this invite
          // and the upsert above will be a no-op due to ignoreDuplicates. Acceptable per Open Q1.
          throw new Error(`reconcileInvitesForUser: invite accept update failed for ${inv.id}: ${inviteError.message}`)
        }
        invitesAccepted += 1
      }

      // 3. Seed first-org for cookie/user_active_org.
      const firstOrgId = invites[0].org_id
      const { data: org } = await admin
        .from('organizations')
        .select('id, name')
        .eq('id', firstOrgId)
        .single()

      // 4. Upsert user_active_org (also service-role; new user has no get_current_org_id).
      await admin
        .from('user_active_org')
        .upsert({ user_id: args.userId, organization_id: firstOrgId, updated_at: now })

      return {
        membershipsCreated,
        invitesAccepted,
        firstOrgId,
        firstOrgName: org?.name ?? null,
      }
    }
    ```

    // Shared private helper — verifies current user is admin of active org.

    async function requireAdminOfActiveOrg() {
      const user = await getUser()

      const supabase = await createClient()
