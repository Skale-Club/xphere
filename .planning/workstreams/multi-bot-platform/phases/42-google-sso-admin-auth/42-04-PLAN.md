'use server'

/**
 * Server actions for the /dashboard/members admin surface (Phase 42).
 *
 * Pattern (from Phase 36 agents/actions.ts):
 *   1. await getUser() — refuse if unauthenticated
 *   2. await createClient() — RLS auto-scopes to active org via get_current_org_id()
 *   3. Mutations end with revalidatePath('/dashboard/members')
 *   4. Return discriminated union { ok: true, data? } | { ok: false, error }
 *
 * RLS contract from migration 045:
 *   - org_invites SELECT/INSERT/UPDATE/DELETE all scoped to active org
 *   - org_members SELECT scoped to active org (or self) via migration 007 policy
 *
 * Service-role escape hatches:
 *   - getMembers uses createAdminClient ONLY to read auth.users.email (which is
 *     not exposed via the public schema). The user_id filter still constrains
 *     the read to memberships of the current org.
 */

import { revalidatePath } from 'next/cache'
import { createClient, getUser } from '@/lib/supabase/server'
import { createAdminClient } from '@/lib/supabase/admin'
import { inviteSchema, type InviteInput } from '@/lib/auth/zod-schemas'
import { normalizeEmail } from '@/lib/auth/normalize-email'
import type { Database } from '@/types/database'

type ActionResult<T = void> =
  | { ok: true; data?: T }
  | { ok: false; error: string }

export interface MemberRow {
  user_id: string
  email: string
  role: 'admin' | 'member'
  created_at: string
}

export interface InviteRow {
  id: string
  email: string
  role: 'admin' | 'member'
  invited_at: string
  expires_at: string | null
  accepted_at: string | null
}

/**
 * List current members of the active org. JOINs org_members with auth.users
 * to surface email (auth.users is outside the public schema → service-role).
 */
export async function getMembers(): Promise<MemberRow[]> {
  const user = await getUser()
  if (!user) return []
  const supabase = await createClient()

  // Step A: get the active org_id via RLS-scoped query (resolves to current).
  const { data: rows, error } = await supabase
    .from('org_members')
    .select('user_id, role, created_at, organization_id')
  if (error || !rows || rows.length === 0) return []

  const orgId = rows[0].organization_id // all rows share active org (RLS)
  const userIds = rows.map((r) => r.user_id)

  // Step B: service-role lookup for emails (auth.users is not in public schema).
  const admin = createAdminClient()
  const { data: emails } = await admin.auth.admin.listUsers({ perPage: 200 })
  const emailMap = new Map<string, string>(
    (emails?.users ?? []).map((u) => [u.id, u.email ?? ''])
  )

  return rows
    .filter((r) => r.organization_id === orgId && userIds.includes(r.user_id))
    .map((r) => ({
      user_id: r.user_id,
      email: emailMap.get(r.user_id) ?? '(unknown)',
      role: r.role,
      created_at: r.created_at,
    }))
    .sort((a, b) => a.created_at.localeCompare(b.created_at))
}

/**
 * List non-accepted, non-expired invites for the active org.
 * RLS scopes to org_id = get_current_org_id().
 */
export async function getPendingInvites(): Promise<InviteRow[]> {
  const user = await getUser()
  if (!user) return []
  const supabase = await createClient()
  const { data } = await supabase
    .from('org_invites')
    .select('id, email, role, invited_at, expires_at, accepted_at')
    .is('accepted_at', null)
    .order('invited_at', { ascending: false })
  return data ?? []
}

/**
 * Insert an invite. Normalizes email (D-42-05). Refuses if the email is
 * already a member of the active org.
 */
export async function inviteMember(raw: InviteInput): Promise<ActionResult> {
  const user = await getUser()
  if (!user) return { ok: false, error: 'Not authenticated' }

  const parsed = inviteSchema.safeParse(raw)
  if (!parsed.success) {
    return { ok: false, error: parsed.error.issues[0]?.message ?? 'Invalid input' }
  }
  const payload = parsed.data
  const email = payload.email // already normalized by emailSchema transform

  const supabase = await createClient()

  // Pre-check: is this email already a member of the active org?
  const admin = createAdminClient()
  const { data: existingUsers } = await admin.auth.admin.listUsers({ perPage: 200 })
  const matchingUser = (existingUsers?.users ?? []).find(
    (u) => normalizeEmail(u.email ?? '') === email
  )
  if (matchingUser) {
    const { data: existingMember } = await supabase
      .from('org_members')
      .select('id')
      .eq('user_id', matchingUser.id)
      .limit(1)
    if (existingMember && existingMember.length > 0) {
      return { ok: false, error: `${email} is already a member of this org` }
    }
  }

  // Resolve active org_id via a scoped read.
  const { data: scoped } = await supabase.from('org_members').select('organization_id').limit(1)
  const orgId = scoped?.[0]?.organization_id
  if (!orgId) return { ok: false, error: 'Could not resolve active organization' }

  const expiresAt = payload.expiresInDays
    ? new Date(Date.now() + payload.expiresInDays * 24 * 60 * 60 * 1000).toISOString()
    : undefined // DB default = now() + 7 days

  const insert: Database['public']['Tables']['org_invites']['Insert'] = {
    org_id: orgId,
    email,
    role: payload.role,
    invited_by: user.id,
    ...(expiresAt && { expires_at: expiresAt }),
  }

  const { error } = await supabase.from('org_invites').insert(insert)
  if (error) {
    // 23505 = unique_violation on (org_id, email)
    if (error.code === '23505') return { ok: false, error: 'An invite for this email already exists' }
    return { ok: false, error: error.message }
  }

  revalidatePath('/dashboard/members')
  return { ok: true }
}

/**
 * Revoke a pending (non-accepted) invite. Accepted invites are immutable
 * audit records — to "uninvite" an accepted user, use removeMember().
 */
export async function revokeInvite(inviteId: string): Promise<ActionResult> {
  const user = await getUser()
  if (!user) return { ok: false, error: 'Not authenticated' }
  const supabase = await createClient()
  const { error } = await supabase
    .from('org_invites')
    .delete()
    .eq('id', inviteId)
    .is('accepted_at', null)
  if (error) return { ok: false, error: error.message }
  revalidatePath('/dashboard/members')
  return { ok: true }
}

/**
 * Remove a member from the active org. Refuses self-removal (D-42-09 — avoid
 * locking yourself out). Session purge is NOT performed; next page load
 * triggers RLS denial and effective logout.
 */
export async function removeMember(userId: string): Promise<ActionResult> {
  const user = await getUser()
  if (!user) return { ok: false, error: 'Not authenticated' }
  if (userId === user.id) {
    return { ok: false, error: 'You cannot remove yourself from the organization' }
  }
  const supabase = await createClient()
  const { error } = await supabase
    .from('org_members')
    .delete()
    .eq('user_id', userId)
  if (error) return { ok: false, error: error.message }
  revalidatePath('/dashboard/members')
  return { ok: true }
}

'use client'

import { useTransition } from 'react'
import { useForm } from 'react-hook-form'
import { zodResolver } from '@hookform/resolvers/zod'
import { toast } from 'sonner'
import { Loader2 } from 'lucide-react'

import { Dialog, DialogContent, DialogHeader, DialogTitle, DialogFooter } from '@/components/ui/dialog'
import { Form, FormField, FormItem, FormLabel, FormControl, FormMessage } from '@/components/ui/form'
import { Input } from '@/components/ui/input'
import { Button } from '@/components/ui/button'
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from '@/components/ui/select'
import { inviteSchema, type InviteInput } from '@/lib/auth/zod-schemas'
import { inviteMember } from '@/app/(dashboard)/members/actions'

interface Props {
  open: boolean
  onOpenChange: (v: boolean) => void
}

export function InviteMemberDialog({ open, onOpenChange }: Props) {
  const [pending, startTransition] = useTransition()
  const form = useForm<InviteInput>({
    resolver: zodResolver(inviteSchema),
    defaultValues: { email: '', role: 'member' },
  })

  function onSubmit(values: InviteInput) {
    startTransition(async () => {
      const r = await inviteMember(values)
      if (r.ok) {
        toast.success('Invite created')
        form.reset()
        onOpenChange(false)
      } else {
        toast.error(r.error)
      }
    })
  }

  return (
    <Dialog open={open} onOpenChange={onOpenChange}>
      <DialogContent>
        <DialogHeader>
          <DialogTitle>Invite a member</DialogTitle>
        </DialogHeader>
        <Form {...form}>
          <form onSubmit={form.handleSubmit(onSubmit)} className="space-y-4">
            <FormField control={form.control} name="email" render={({ field }) => (
              <FormItem>
                <FormLabel>Email</FormLabel>
                <FormControl><Input type="email" placeholder="person@example.com" {...field} /></FormControl>
                <FormMessage />
              </FormItem>
            )} />
            <FormField control={form.control} name="role" render={({ field }) => (
              <FormItem>
                <FormLabel>Role</FormLabel>
                <Select value={field.value} onValueChange={field.onChange}>
                  <FormControl><SelectTrigger><SelectValue /></SelectTrigger></FormControl>
                  <SelectContent>
                    <SelectItem value="admin">Admin</SelectItem>
                    <SelectItem value="member">Member</SelectItem>
                  </SelectContent>
                </Select>
                <FormMessage />
              </FormItem>
            )} />
            <DialogFooter>
              <Button type="button" variant="outline" onClick={() => onOpenChange(false)}>Cancel</Button>
              <Button type="submit" disabled={pending}>
                {pending && <Loader2 className="mr-2 h-4 w-4 animate-spin" />}
                Create invite
              </Button>
            </DialogFooter>
          </form>
        </Form>
        <p className="text-xs text-muted-foreground">
          The invitee can sign in via Google. Share the /login URL with them — Phase 42 v1 does not send email automatically.
        </p>
      </DialogContent>
    </Dialog>
  )
}
