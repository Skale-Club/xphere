/**
 * Canonical email normalization for invite lookups + OAuth callback matching.
 *
 * Phase 42 contract (D-42-05): lowercase + trim only. We deliberately do NOT
 * strip plus-addressing (`user+tag@gmail.com`) or apply Gmail's dot-removal —
 * those are provider-specific heuristics that would silently collapse two
 * distinct invites into one. Admins explicitly type the email; we trust it.
 *
 * Use this helper in BOTH the OAuth callback lookup AND the invite INSERT path
 * (Plan 04 `inviteMember` action) so the two sides always normalize identically.
 */
export function normalizeEmail(raw: string): string {
  return raw.trim().toLowerCase()
}
```

/**
 * Service-role Supabase client. Bypasses RLS — use ONLY in trusted server
 * contexts that cannot rely on the authenticated user's RLS scope.
 *
 * Phase 42 (D-42-04): the OAuth callback consumes invites for a user who has
 * just been created and has no org_members row yet, so `get_current_org_id()`
 * cannot resolve their org. Service-role lets the callback look up invites
 * across orgs by normalized email and insert org_members on their behalf.
 *
 * Do NOT export this from any client-side module. The service role key must
 * never leave the Node.js runtime.
 */
import { createClient as createSupabaseClient } from '@supabase/supabase-js'
import type { Database } from '@/types/database'

export function createAdminClient() {
  const url = process.env.NEXT_PUBLIC_SUPABASE_URL
  const serviceKey = process.env.SUPABASE_SERVICE_ROLE_KEY

  if (!url) throw new Error('NEXT_PUBLIC_SUPABASE_URL is not set')
  if (!serviceKey) {
    throw new Error(
      'SUPABASE_SERVICE_ROLE_KEY is not set — required by OAuth callback for invite consumption (Phase 42)'
    )
  }

  return createSupabaseClient<Database>(url, serviceKey, {
    auth: { persistSession: false, autoRefreshToken: false },
  })
}
```

```ts
/**
 * OAuth callback for Google SSO (Phase 42).
 *
 * Flow:
 *   1. Exchange `?code=` for a Supabase session (cached server client).
 *   2. Read the just-authenticated user via getUser() cached helper.
 *   3. Normalize the email (lowercase + trim — D-42-05).
 *   4. With service-role: SELECT non-expired, non-accepted org_invites rows
 *      whose lower(email) matches.
 *   5. For each match: INSERT org_members ON CONFLICT DO NOTHING (multi-org
 *      friendly — D-42-06) + UPDATE org_invites SET accepted_at = now().
 *   6. Determine target org for the cookie: first row in org_members for
 *      this user (covers both new matches and returning legacy users).
 *   7. If still zero memberships → signOut + redirect /login?error=not_invited.
 *   8. Otherwise set vo_active_org cookie and redirect /dashboard.
 *
 * Why service-role for steps 4-5: the user has no org context yet, so RLS
 * (which keys off get_current_org_id()) cannot authorize the cross-org
 * SELECT or the org_members INSERT. Service-role is scoped to this single
 * trusted server file (D-42-04, D-42-14).
 */
import { NextResponse, type NextRequest } from 'next/server'
import { createClient, getUser } from '@/lib/supabase/server'
import { createAdminClient } from '@/lib/supabase/admin'
import { normalizeEmail } from '@/lib/auth/normalize-email'

export const runtime = 'nodejs'
export const dynamic = 'force-dynamic'

export async function GET(request: NextRequest) {
  const url = new URL(request.url)
  const code = url.searchParams.get('code')
  const origin = url.origin

  if (!code) {
    return NextResponse.redirect(`${origin}/login?error=missing_code`)
  }

  // Step 1: exchange code for session (writes auth cookies via server client)
  const supabase = await createClient()
  const { error: exchangeError } = await supabase.auth.exchangeCodeForSession(code)
  if (exchangeError) {
    console.error('[auth/callback] exchangeCodeForSession failed', exchangeError)
    return NextResponse.redirect(`${origin}/login?error=oauth_failed`)
  }

  // Step 2: read authenticated user
  const user = await getUser()
  if (!user || !user.email) {
    return NextResponse.redirect(`${origin}/login?error=oauth_failed`)
  }

  const normalized = normalizeEmail(user.email)
  const admin = createAdminClient()

  // Step 4: match invites
  const nowIso = new Date().toISOString()
  const { data: invites, error: invErr } = await admin
    .from('org_invites')
    .select('id, org_id, role')
    .ilike('email', normalized) // belt-and-suspenders; invites are stored normalized too
    .is('accepted_at', null)
    .or(`expires_at.is.null,expires_at.gt.${nowIso}`)

  if (invErr) {
    console.error('[auth/callback] invite lookup failed', invErr)
    return NextResponse.redirect(`${origin}/login?error=oauth_failed`)
  }

  // Step 5: consume matches (insert membership + mark accepted)
  for (const inv of invites ?? []) {
    const { error: memErr } = await admin
      .from('org_members')
      .upsert(
        { user_id: user.id, organization_id: inv.org_id, role: inv.role },
        { onConflict: 'user_id,organization_id', ignoreDuplicates: true }
      )
    if (memErr) {
      console.error('[auth/callback] org_members insert failed', { invite: inv.id, memErr })
      continue
    }
    await admin
      .from('org_invites')
      .update({ accepted_at: nowIso })
      .eq('id', inv.id)
  }

  // Step 6: resolve active org (first membership wins for cookie)
  const { data: memberships } = await admin
    .from('org_members')
    .select('organization_id, created_at')
    .eq('user_id', user.id)
    .order('created_at', { ascending: true })
    .limit(1)

  // Step 7: not_invited gate
  if (!memberships || memberships.length === 0) {
    await supabase.auth.signOut()
    return NextResponse.redirect(`${origin}/login?error=not_invited`)
  }

  // Step 8: set cookie + redirect to dashboard
  const response = NextResponse.redirect(`${origin}/`)
  response.cookies.set('vo_active_org', memberships[0].organization_id, {
    httpOnly: false, // OrgSwitcher reads this client-side — preserve existing convention
    sameSite: 'lax',
    secure: process.env.NODE_ENV === 'production',
    path: '/',
    maxAge: 60 * 60 * 24 * 30, // 30 days
  })

  // Best-effort mirror to user_active_org table (RLS allows authenticated insert/update on self)
  // Failures are non-fatal — cookie is the authoritative selection in v1.7+.
  await supabase
    .from('user_active_org')
    .upsert({ user_id: user.id, organization_id: memberships[0].organization_id })

  return response
}
```
