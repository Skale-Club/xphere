import { describe, it, expect } from 'vitest'
import { normalizeEmail } from '@/lib/auth/normalize-email'
import { emailSchema, inviteSchema, roleEnum } from '@/lib/auth/zod-schemas'

describe('normalizeEmail', () => {
  it('lowercases and trims', () => {
    expect(normalizeEmail('  John@Example.COM\n')).toBe('john@example.com')
  })

  it('returns empty string for empty input', () => {
    expect(normalizeEmail('')).toBe('')
  })

  it('preserves plus-addressing (no Gmail-specific dot stripping)', () => {
    expect(normalizeEmail('User+Tag@Gmail.com')).toBe('user+tag@gmail.com')
  })

  it('trims tabs and newlines', () => {
    expect(normalizeEmail('\t\nfoo@bar.com\t')).toBe('foo@bar.com')
  })

  it('handles unicode email components', () => {
    expect(normalizeEmail('  Bjørn@Example.com ')).toBe('bjørn@example.com')
  })

  it('is idempotent', () => {
    const n = normalizeEmail('  A@B.com ')
    expect(normalizeEmail(n)).toBe(n)
  })
})

describe('emailSchema', () => {
  it('rejects non-emails', () => {
    expect(emailSchema.safeParse('not-an-email').success).toBe(false)
  })
  it('normalizes valid emails', () => {
    const r = emailSchema.safeParse('  USER@Example.com ')
    expect(r.success).toBe(true)
    expect(r.success && r.data).toBe('user@example.com')
  })
})

describe('roleEnum', () => {
  it('accepts admin and member', () => {
    expect(roleEnum.safeParse('admin').success).toBe(true)
    expect(roleEnum.safeParse('member').success).toBe(true)
  })
  it('rejects unknown roles per D-42-07', () => {
    expect(roleEnum.safeParse('owner').success).toBe(false)
    expect(roleEnum.safeParse('viewer').success).toBe(false)
  })
})

describe('inviteSchema', () => {
  it('defaults role to member', () => {
    const r = inviteSchema.safeParse({ email: 'a@b.com' })
    expect(r.success).toBe(true)
    expect(r.success && r.data.role).toBe('member')
  })
  it('rejects negative or absurd expiresInDays', () => {
    expect(inviteSchema.safeParse({ email: 'a@b.com', role: 'admin', expiresInDays: 0 }).success).toBe(false)
    expect(inviteSchema.safeParse({ email: 'a@b.com', role: 'admin', expiresInDays: 9999 }).success).toBe(false)
  })
  it('accepts valid expiresInDays', () => {
    const r = inviteSchema.safeParse({ email: 'a@b.com', role: 'admin', expiresInDays: 30 })
    expect(r.success).toBe(true)
  })
})

 * Phase 42 RLS isolation — AUTH-07.
 *
 * Creates two orgs each with an admin user; asserts that User X's
 * authenticated client cannot see/touch User Y's org's invites or members,
 * and vice versa.
 *
 * Fixture prefix: phase42-rls-* (matches Phase 36 TEST_FIXTURE_NAME regex).

import { describe, it, expect, beforeAll, afterAll } from 'vitest'
import { createClient as createBrowserClient } from '@supabase/supabase-js'
import { createAdminClient } from '@/lib/supabase/admin'
import type { Database } from '@/types/database'

const FIXTURE_PREFIX = 'phase42-rls'
const PW = 'TestPass123!'

const admin = createAdminClient()
const url = process.env.NEXT_PUBLIC_SUPABASE_URL!
const anon = process.env.NEXT_PUBLIC_SUPABASE_PUBLISHABLE_KEY!

const state = {
  orgX: '',
  orgY: '',
  userX: { id: '', email: '' },
  userY: { id: '', email: '' },
  inviteX: '',
  inviteY: '',
  clientX: null as ReturnType<typeof createBrowserClient<Database>> | null,
  clientY: null as ReturnType<typeof createBrowserClient<Database>> | null,
}

async function signIn(email: string) {
  const c = createBrowserClient<Database>(url, anon)
  const { error } = await c.auth.signInWithPassword({ email, password: PW })
  if (error) throw new Error(`signIn ${email}: ${error.message}`)
  return c
}

async function setActiveOrg(c: NonNullable<typeof state.clientX>, userId: string, orgId: string) {
  // Mirror what the OAuth callback does: write to user_active_org.
  // get_current_org_id() reads from this table first.
  await c.from('user_active_org').upsert({ user_id: userId, organization_id: orgId })
}
