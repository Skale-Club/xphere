    ```typescript
    export async function fetchBusyTimes(
      userId: string,
      orgId: string,
      timeMin: string,
      timeMax: string,
      calendarIds: string[] = ['primary'],
    ): Promise<BusyInterval[]> {
      const tokens = await getCalendarTokens(userId, orgId)
      if (!tokens) return []

      const ids = calendarIds.length > 0 ? calendarIds : ['primary']

      const body = {
        timeMin,
        timeMax,
        items: ids.map((id) => ({ id })),
      }

      const res = await fetch('https://www.googleapis.com/calendar/v3/freeBusy', {
        method: 'POST',
        headers: {
          Authorization: `Bearer ${tokens.access_token}`,
          'Content-Type': 'application/json',
        },
        body: JSON.stringify(body),
        cache: 'no-store',
      })

      if (!res.ok) {
        console.error('[google-calendar] freeBusy error:', res.status)
        return []
      }

      const data = (await res.json()) as {
        calendars?: Record<string, { busy?: BusyInterval[] }>
      }

      return ids.flatMap((id) => data.calendars?.[id]?.busy ?? [])
    }
    ```
