1. **Silent coercion for TRULY unrecognized statuses remains.** Post-127-05, `mapStatus`'s fallback (`return 'confirmed'`) still silently absorbs ANY string that isn't `cancelled`/`no_show`/`completed` — including a typo, a new Xkedule status this codebase doesn't know about yet, or a malformed payload. `pending`/`awaiting_approval` intentionally fall back to `confirmed` (documented, acceptable — there is no native "pending" state), but a genuinely unrecognized value doing the same thing is exactly the anti-pattern SYNC-02/D-02 forbids ("no silent coercion"). 2. **Existing-row status updates still write `bookings.status` directly.** 127-05 keeps `await supabase.from('bookings').update(mutable).eq('id', existing.id)` — `mutable` still includes the mapped `status` literal — as a raw write. SYNC-02/D-02 requires "the Phase 127 canonical lifecycle service — same transitions, same events, no direct status writes." This plan splits that update: non-status fields go through a plain `.update()` as before; the STATUS field is now routed through `src/lib/calendar/transition.ts`'s `confirmBooking`/`cancelBooking`/`markNoShow`/`markShowed` (Plan 127-01's output — confirmed signature `(ctx, bookingId, orgId) => Promise<{ ok: boolean; error?: string }>`, idempotent no-op with no re-emitted event if the booking is already at the target status).

import type { BookingStatus } from '@/lib/calendar/booking-status'

function mapStatus(s: string): BookingStatus {
  if (s === 'cancelled') return 'cancelled'
  if (s === 'no_show') return 'no_show'
  if (s === 'completed') return 'showed'
  return 'confirmed'   // pending, awaiting_approval, AND any unrecognized value all land here
}

function calendarEventFor(event: string, status: BookingStatus): CalendarEvent {
  if (event === 'booking.cancelled' || status === 'cancelled') return 'meeting.cancelled'
  if (status === 'no_show') return 'meeting.no_show'
  if (status === 'showed') return 'meeting.completed'
  if (event === 'booking.created') return 'meeting.scheduled'
  if (event === 'booking.confirmed') return 'meeting.confirmed'
  return 'meeting.rescheduled'
}
