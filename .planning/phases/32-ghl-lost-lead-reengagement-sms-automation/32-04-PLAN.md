// src/app/api/automations/ghl-reengagement/run/route.ts
// Phase 32 (v1.9): Protected internal endpoint that runs one pass of the
// GHL Lost-Lead Reengagement SMS automation. Triggered by GitHub Actions cron
// or manual workflow_dispatch. NOT a public webhook — bearer-secret protected.
//
// Pattern source: 32-RESEARCH.md Pattern 7 (bearer auth) + Pattern 3 (service-role
// client) + Code Examples — Loading the GHL integration row.
//
// Unlike /api/vapi/tools/route.ts (Vapi webhook — always 200), this endpoint
// returns 401 on auth failure and 500 on env / runtime errors so the GitHub
// Action surfaces them as failed workflow runs.

import { timingSafeEqual } from 'node:crypto'
import { createServiceRoleClient } from '@/lib/supabase/admin'
import { decrypt } from '@/lib/crypto'
import { runReengagement } from '@/lib/automations/ghl-reengagement/runner'
import type { GhlCredentials } from '@/lib/ghl/client'

export const runtime = 'nodejs'

// Locked planning constants (see 32-04-PLAN.md "planning_decisions_locked")
const DEFAULT_THRESHOLD_DAYS = 180
const DEFAULT_BATCH_LIMIT = 20

const REQUIRED_ENV = [
  'GHL_REENGAGEMENT_LOCATION_ID',
  'GHL_REENGAGEMENT_INTEGRATION_ID',
  'GHL_REENGAGEMENT_TWILIO_INTEGRATION_ID',
  'GHL_REENGAGEMENT_MESSAGE',
  'GHL_REENGAGEMENT_TRIGGER_SECRET',
] as const

function isAuthorized(request: Request): boolean {
  const header = request.headers.get('authorization') ?? ''
  const m = header.match(/^Bearer\s+(.+)$/)
  if (!m) return false
  const expected = process.env.GHL_REENGAGEMENT_TRIGGER_SECRET ?? ''
  if (!expected) return false
  const providedBuf = Buffer.from(m[1])
  const expectedBuf = Buffer.from(expected)
  if (providedBuf.length !== expectedBuf.length) return false
  return timingSafeEqual(providedBuf, expectedBuf)
}

function jsonError(message: string, status: number): Response {
  return Response.json({ error: message }, { status })
}

export async function POST(request: Request): Promise<Response> {
  // 1. Auth — bearer secret with constant-time compare
  if (!isAuthorized(request)) {
    return jsonError('unauthorized', 401)
  }

  // 2. Required env vars
  for (const name of REQUIRED_ENV) {
    if (!process.env[name]) {
      return jsonError(`missing required env var: ${name}`, 500)
    }
  }

  // 3. Optional env vars with defaults
  const thresholdDays = (() => {
    const raw = process.env.GHL_REENGAGEMENT_THRESHOLD_DAYS
    if (!raw) return DEFAULT_THRESHOLD_DAYS
    const n = Number.parseInt(raw, 10)
    return Number.isInteger(n) && n > 0 ? n : DEFAULT_THRESHOLD_DAYS
  })()
  const batchLimit = (() => {
    const raw = process.env.GHL_REENGAGEMENT_BATCH_LIMIT
    if (!raw) return DEFAULT_BATCH_LIMIT
    const n = Number.parseInt(raw, 10)
    return Number.isInteger(n) && n > 0 ? n : DEFAULT_BATCH_LIMIT
  })()

  try {
    const supabase = createServiceRoleClient()

    // 4. Load GHL integration row + decrypt API key
    const { data: row, error: rowErr } = await supabase
      .from('integrations')
      .select('id, organization_id, encrypted_api_key, is_active')
      .eq('id', process.env.GHL_REENGAGEMENT_INTEGRATION_ID!)
      .eq('provider', 'gohighlevel')
      .eq('is_active', true)
      .single()

    if (rowErr || !row) {
      return jsonError(
        'GHL integration not found or inactive (check GHL_REENGAGEMENT_INTEGRATION_ID)',
        500,
      )
    }

    const apiKey = await decrypt(row.encrypted_api_key)
    const credentials: GhlCredentials = {
      apiKey,
      locationId: process.env.GHL_REENGAGEMENT_LOCATION_ID!,
    }

    // 5. Run one pass
    const result = await runReengagement(
      {
        orgId: row.organization_id,
        locationId: process.env.GHL_REENGAGEMENT_LOCATION_ID!,
        ghlCredentials: credentials,
        twilioIntegrationId: process.env.GHL_REENGAGEMENT_TWILIO_INTEGRATION_ID!,
        messageTemplate: process.env.GHL_REENGAGEMENT_MESSAGE!,
        thresholdDays,
        batchLimit,
        runStartedAtIso: new Date().toISOString(),
      },
      supabase,
    )

    return Response.json(result, { status: 200 })
  } catch (err) {
    const message = err instanceof Error ? err.message : String(err)
    // NEVER echo Authorization header value. message comes from runner / supabase / decrypt — sanitized at source.
    return jsonError(message, 500)
  }
}

# GHL Lost-Lead Reengagement (SMS) — Operator Setup

**Phase:** 32 (milestone v1.9)
**Automation type:** scheduled outbound (cron-only in v1.9)
**Target:** Skleanings GHL sub-account; one SMS per Lost opportunity > 180 days (configurable) — never re-sent.

***

## Overview

A daily GitHub Action POSTs to a protected internal endpoint on
`https://operator.skale.club/api/automations/ghl-reengagement/run`. The endpoint:

1. Lists GHL Lost opportunities older than `THRESHOLD_DAYS`
2. Skips contacts already in the `ghl_reengagement_sent` anti-loop table
3. Renders the SMS body by substituting `{{first_name}}` (falls back to `amigo(a)`)
4. Claim-first INSERT into `ghl_reengagement_sent`, then dispatches via Twilio
5. On Twilio failure, rolls back the claim (so the contact retries next run)
6. Logs every attempt in `action_logs` with `tool_name='ghl_reengagement_sms'`

The endpoint returns JSON: `{ processed, sent, skipped, failed, errors[] }`.

***

## Required Vercel Environment Variables

Set these on the **production** scope of the Vercel project:

| Variable | Required | Description |
|----------|----------|-------------|
| `GHL_REENGAGEMENT_LOCATION_ID` | yes | GHL sub-account (location) ID for Skleanings |
| `GHL_REENGAGEMENT_INTEGRATION_ID` | yes | `integrations.id` (UUID) for the row holding the Skleanings GHL Private Integration Token |
| `GHL_REENGAGEMENT_TWILIO_INTEGRATION_ID` | yes | `integrations.id` (UUID) for the Skleanings Twilio integration row (must match the org's active Twilio integration) |
| `GHL_REENGAGEMENT_MESSAGE` | yes | SMS template — must contain `{{first_name}}` placeholder (e.g., `"Olá {{first_name}}, sentimos sua falta na Skleanings! Quer agendar uma limpeza?"`) |
| `GHL_REENGAGEMENT_TRIGGER_SECRET` | yes | Bearer secret. Generate via `openssl rand -hex 32`. The same value goes into the GitHub secret below. |
| `GHL_REENGAGEMENT_THRESHOLD_DAYS` | no (default `180`) | Only opportunities whose `updatedAt` is older than this many days are eligible |
| `GHL_REENGAGEMENT_BATCH_LIMIT` | no (default `20`) | Max successful sends per cron tick. Default sized for Vercel Hobby 10s function timeout. Raise to 50–100 only if on Vercel Pro. |

## Cron Schedule

The workflow is defined in `.github/workflows/ghl-reengagement.yml`:

```yaml
on:
  schedule:
    - cron: '0 14 * * *'
  workflow_dispatch:
```

`0 14 * * *` = **14:00 UTC daily** ≈ **11:00 BRT** (BRT = UTC-3; ignores DST).

GitHub Actions does not guarantee exact cron timing — expect drift of up to 15 minutes during peak load.

To run the automation ad-hoc:

1. Open the GitHub repository in a browser
2. Go to **Actions** tab
3. Select **GHL Lost-Lead Reengagement (SMS)** in the workflow list
4. Click **Run workflow** (top right)
5. Confirm — the run uses the same env vars / secrets as the scheduled run

You can also trigger via `gh` CLI:

```sh
gh workflow run ghl-reengagement.yml
```

## Operational Verification

After a run completes (scheduled or manual):

1. **GitHub Actions log:** the workflow's `POST runner endpoint` step prints `HTTP Status: 200` and the JSON body
2. **`action_logs` table:** every dispatch attempt appears with `tool_name='ghl_reengagement_sms'` and `vapi_call_id='cron:ghl-reengagement:<iso>'`
3. **`ghl_reengagement_sent` table:** one row per contact that received an SMS this run (or earlier)

Query in Supabase Studio:

```sql
SELECT vapi_call_id, count(*), sum(case when status='success' then 1 else 0 end) as ok
FROM action_logs
WHERE tool_name = 'ghl_reengagement_sms'
GROUP BY vapi_call_id
ORDER BY 1 DESC LIMIT 10;
```

## Troubleshooting

| Symptom | Likely Cause | Fix |
|---------|--------------|-----|
| Workflow shows HTTP 401 | Bearer mismatch between GitHub secret and Vercel env | Reset both to the same `openssl rand -hex 32` value |
| Workflow shows HTTP 500, body `missing required env var: X` | Env var `X` not set on Vercel production scope | Add via Vercel dashboard, redeploy |
| Workflow shows HTTP 500, body mentions Twilio integration mismatch | `GHL_REENGAGEMENT_TWILIO_INTEGRATION_ID` UUID doesn't match the org's active Twilio row | Look up the correct `integrations.id` in Supabase and update Vercel env |
| Many `action_logs` with status='error' and `Twilio error 21211` | Contact phone not in E.164 format | Clean phone numbers in GHL; runner counts these as `skipped`, not `failed` |
| Runs but `sent: 0` and `processed: 0` | All Lost opportunities younger than threshold OR GHL date filter param name is wrong | Lower `GHL_REENGAGEMENT_THRESHOLD_DAYS` to test; if still 0 and you expect data, change `GHL_DATE_FILTER_PARAM` constant in `src/lib/ghl/list-opportunities.ts` to `endDate` or `lastStatusChangeStartDate` (see 32-RESEARCH.md Pitfall 1) |
| Vercel function timeout in logs | `GHL_REENGAGEMENT_BATCH_LIMIT` set too high for current Vercel plan | Lower to 20 (Hobby) or 50 (Pro default) |
